function manageCoffee() {
  while (true) {
    let coffee = checkCoffeeSituation();
    let codingMood = "focused";
    let productivityLevel = 0;

    switch (coffee) {
      case "empty":
        console.log("☕ Waktunya isi ulang kopi!");
        brewFreshCoffee({ type: "Arabica", intensity: "strong" });
        productivityLevel += 20;
        break;

        drinkCoffee({ sips: "slow", enjoyment: "maximum" });
        codingMood = "supercharged";
        productivityLevel += 50;
        break;

        continue;

        brewTea({ type: "green", honey: true });
        productivityLevel += 10;
    }

    return {
      mood: codingMood,
      productivity: `${productivityLevel}%`,
      nextAction: "Write clean, efficient blockchain code ⛓️",
    };
  }
}
